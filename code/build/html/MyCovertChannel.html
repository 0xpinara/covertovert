Enhanced Covert Storage Channel implementation using ARP Destination MAC Address field.

BONUS Implementation: - Uses last two bytes of MAC address to encode 2 bits per packet - First 4 bytes remain constant (00:11:22:33) for packet identification - Bytes 5-6 encode two bits using the pattern:

  • 00:00 for ‘00’

  • 00:FF for ‘01’

  • FF:00 for ‘10’

  • FF:FF for ‘11’

Channel Capacity: ~16.36 bits/second

  6class MyCovertChannel(CovertChannelBase):
  7    """
  8    Enhanced Covert Storage Channel implementation using ARP Destination MAC Address field.
  9    
 10    BONUS Implementation:
 11    - Uses last two bytes of MAC address to encode 2 bits per packet
 12    - First 4 bytes remain constant (00:11:22:33) for packet identification
 13    - Bytes 5-6 encode two bits using the pattern:
 14      - 00:00 for '00'
 15      - 00:FF for '01'
 16      - FF:00 for '10'
 17      - FF:FF for '11'
 18    
 19    Channel Capacity: ~16.36 bits/second 
 20    """

 22    def __init__(self):
 23        super().__init__()
 24        conf.verb = 0

 26    def encode_bits_to_mac(self, bits):
 27        """Helper function to encode 2 bits into a MAC address"""
 28        base_mac = "00:11:22:33:"  # First 4 bytes constant
 29        # Encode 2 bits using last two bytes
 30        if bits == "00":
 31            return base_mac + "00:00"
 32        elif bits == "01":
 33            return base_mac + "00:FF"
 34        elif bits == "10":
 35            return base_mac + "FF:00"
 36        else:  # "11"
 37            return base_mac + "FF:FF"
 38    
 39    def decode_mac_to_bits(self, mac):
 40        """Helper function to decode a MAC address into 2 bits"""
 41        last_two_bytes = mac.split(":")[-2:]
 42        byte5, byte6 = [int(b, 16) for b in last_two_bytes]
 43        
 44        if byte5 == 0 and byte6 == 0:
 45            return "00"
 46        elif byte5 == 0 and byte6 == 255:
 47            return "01"
 48        elif byte5 == 255 and byte6 == 0:
 49            return "10"
 50        else:  # byte5 == 255 and byte6 == 255
 51            return "11"

 53    def send(self, min_length, max_length, log_file_name, packet_delay):
 54        """Send covert message using enhanced ARP packets encoding"""
 55        message = self.generate_random_message(min_length, max_length)
 56        self.log_message(message, log_file_name)
 57        
 58        print(f"Sending message: {message}")
 59        binary_message = ''.join(format(ord(c), '08b') for c in message)
 60        print(f"Binary message: {binary_message}")
 61        
 62        start_time = time.time()
 63        
 64        # Send two bits at a time
 65        for i in range(0, len(binary_message), 2):
 66            bits = binary_message[i:i+2]
 67            if len(bits) == 1:  # Handle odd number of bits
 68                bits += "0"
 69                
 70            arp = Ether(dst="ff:ff:ff:ff:ff:ff")/ARP(
 71                op=1,
 72                hwsrc="02:42:ac:11:00:02",
 73                psrc="172.18.0.2",
 74                hwdst=self.encode_bits_to_mac(bits),
 75                pdst="172.18.0.3"
 76            )
 77            
 78            super().send(arp, interface="eth0")
 79            print(f"Sent bits: {bits}")
 80            time.sleep(packet_delay)
 81        
 82        end_time = time.time()
 83        duration = end_time - start_time
 84        capacity = len(binary_message) / duration
 85        print(f"Enhanced channel capacity: {capacity:.2f} bits/second")

 87    def receive(self, log_file_name, timeout):
 88        """Receive and decode covert message from enhanced ARP packets"""
 89        received_bits = []
 90        print(f"Starting enhanced receiver on eth0 (timeout: {timeout}s)")
 91        
 92        def packet_callback(packet):
 93            if ARP in packet:
 94                # Check if it's our covert channel packet
 95                if packet[ARP].hwdst.startswith("00:11:22:33"):
 96                    mac = packet[ARP].hwdst
 97                    bits = self.decode_mac_to_bits(mac)
 98                    received_bits.extend(list(bits))
 99                    print(f"Received bits: {bits} from MAC: {mac}")
100        
101        try:
102            print("Waiting for packets...")
103            sniff(
104                iface="eth0",
105                filter="arp",
106                prn=packet_callback,
107                timeout=timeout,
108                store=0
109            )
110        except Exception as e:
111            print(f"Error during capture: {e}")
112        
113        print(f"Capture complete. Received {len(received_bits)} bits")
114        
115        if not received_bits:
116            print("No bits received")
117            return
118        
119        # Convert bits to message
120        bits = ''.join(received_bits)
121        print(f"Received bits: {bits}")
122        
123        message = ''
124        for i in range(0, len(bits), 8):
125            byte = bits[i:i+8]
126            if len(byte) == 8:
127                message += chr(int(byte, 2))
128        
129        self.log_message(message, log_file_name)
130        print(f" message: {message}")
